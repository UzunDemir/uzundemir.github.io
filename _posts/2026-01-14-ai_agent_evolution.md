    def optimize_node_prompts(self) -> List[Dict]:
        """
        –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∞—è –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è –ø—Ä–æ–º–ø—Ç–æ–≤ —á–µ—Ä–µ–∑ –≥–µ–Ω–µ—Ç–∏—á–µ—Å–∫–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º
        """
        
        # –°–æ–±–∏—Ä–∞–µ–º —É—Å–ø–µ—à–Ω—ã–µ –∏ –Ω–µ—É—Å–ø–µ—à–Ω—ã–µ –ø—Ä–∏–º–µ—Ä—ã
        training_data = self._collect_prompt_training_data()
        
        optimized_prompts = []
        
        for node_name in self.base_agent.nodes.keys():
            print(f"[–û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è] –û–ø—Ç–∏–º–∏–∑–∏—Ä—É—é –ø—Ä–æ–º–ø—Ç –¥–ª—è —É–∑–ª–∞: {node_name}")
            
            # –¢–µ–∫—É—â–∏–π –ø—Ä–æ–º–ø—Ç
            current_prompt = self.base_agent.get_node_prompt(node_name)
            
            # –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –≤–∞—Ä–∏–∞—Ü–∏–π –ø—Ä–æ–º–ø—Ç–∞
            variations = self._generate_prompt_variations(
                current_prompt, 
                node_name, 
                training_data
            )
            
            # –û—Ü–µ–Ω–∫–∞ –≤–∞—Ä–∏–∞—Ü–∏–π
            scored_variations = []
            for variation in variations:
                score = self._evaluate_prompt_variation(
                    node_name, variation, training_data
                )
                scored_variations.append({
                    "prompt": variation,
                    "score": score,
                    "node": node_name
                })
            
            # –í—ã–±–æ—Ä –ª—É—á—à–µ–≥–æ
            best_variation = max(scored_variations, key=lambda x: x["score"])
            
            # –ü—Ä–æ–≤–µ—Ä–∫–∞, —á—Ç–æ —É–ª—É—á—à–µ–Ω–∏–µ –∑–Ω–∞—á–∏–º–æ
            if best_variation["score"] > self._get_baseline_score(node_name) * 1.05:
                optimized_prompts.append(best_variation)
                print(f"[–û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è] –ù–∞–π–¥–µ–Ω–æ —É–ª—É—á—à–µ–Ω–∏–µ –¥–ª—è {node_name}: +{((best_variation['score']/self._get_baseline_score(node_name))-1)*100:.1f}%")
        
        return optimized_prompts
    
    def _generate_prompt_variations(self, base_prompt: str, node_name: str, training_data: List) -> List[str]:
        """
        –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –≤–∞—Ä–∏–∞—Ü–∏–π –ø—Ä–æ–º–ø—Ç–∞ —Å –ø–æ–º–æ—â—å—é LLM
        """
        
        variation_prompt = f"""
        –¢—ã —ç–∫—Å–ø–µ—Ä—Ç –ø–æ –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏–∏ –ø—Ä–æ–º–ø—Ç–æ–≤ –¥–ª—è –ò–ò-—Å–∏—Å—Ç–µ–º.
        
        –ë–ê–ó–û–í–´–ô –ü–†–û–ú–ü–¢:

        –£–ó–ï–õ –í–´–ü–û–õ–ù–Ø–ï–¢: {self.base_agent.nodes[node_name].task}
        
        –î–ê–ù–ù–´–ï –î–õ–Ø –û–ë–£–ß–ï–ù–ò–Ø:
        –£—Å–ø–µ—à–Ω—ã–µ –ø—Ä–∏–º–µ—Ä—ã: {len([d for d in training_data if d['success']])}
        –ù–µ—É—Å–ø–µ—à–Ω—ã–µ –ø—Ä–∏–º–µ—Ä—ã: {len([d for d in training_data if not d['success']])}

        1. –£—á–∏—Ç—ã–≤–∞—é—Ç —á–∞—Å—Ç—ã–µ –æ—à–∏–±–∫–∏ –∏–∑ –Ω–µ—É—Å–ø–µ—à–Ω—ã—Ö –ø—Ä–∏–º–µ—Ä–æ–≤
        2. –°–æ—Ö—Ä–∞–Ω—è—é—Ç —Å–∏–ª—å–Ω—ã–µ —Å—Ç–æ—Ä–æ–Ω—ã –∏–∑ —É—Å–ø–µ—à–Ω—ã—Ö –ø—Ä–∏–º–µ—Ä–æ–≤
        3. –ë–æ–ª–µ–µ —á—ë—Ç–∫–æ —Ñ–æ—Ä–º—É–ª–∏—Ä—É—é—Ç –∑–∞–¥–∞—á—É

        response = self.llm.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": variation_prompt}],
            temperature=0.7,  # –í—ã—Å–æ–∫–∞—è —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ –¥–ª—è –∫—Ä–µ–∞—Ç–∏–≤–Ω–æ—Å—Ç–∏

        result = json.loads(response.choices[0].message.content)
        return result.get("variations", [])
    
    def ab_test_configurations(self, candidate_configs: Dict) -> Dict:
        """
        –ö–æ–º–ø–ª–µ–∫—Å–Ω–æ–µ A/B —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏–π

        test_cases = self._create_test_suite(n_cases=100)
        
        # –¢–µ–∫—É—â–∞—è –∫–æ–Ω—Ñ–∏–≥—É—Ä–∞—Ü–∏—è (–∫–æ–Ω—Ç—Ä–æ–ª—å–Ω–∞—è –≥—Ä—É–ø–ø–∞)

            config=None,  # –¢–µ–∫—É—â–∞—è
            test_cases=test_cases,
            group_name="baseline"
        )
        
        best_config = None
        best_improvement = 0
        
        # –¢–µ—Å—Ç–∏—Ä—É–µ–º –∫–∞–∂–¥—É—é –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏—é –æ—Ç–¥–µ–ª—å–Ω–æ
        for config_type, configs in candidate_configs.items():
            if not configs:
                continue
                
            print(f"[A/B Test] –¢–µ—Å—Ç–∏—Ä—É—é {config_type}...")
            
            for config in configs:
                test_results = self._test_configuration(
                    config=config,
                    test_cases=test_cases,
                    group_name=f"{config_type}_{hash(str(config))[:8]}"
                )
                
                # –°—Ä–∞–≤–Ω–∏–≤–∞–µ–º —Å baseline
                improvement = self._calculate_improvement(
                    baseline_results, test_results

                print(f"[A/B Test] {config_type}: —É–ª—É—á—à–µ–Ω–∏–µ {improvement*100:.1f}%")
                
                if improvement > best_improvement and improvement > 0.03:  # –ú–∏–Ω–∏–º—É–º 3%
                    best_improvement = improvement
                    best_config = {
                        "type": config_type,
                        "config": config,
                        "improvement": improvement,
                        "test_results": test_results
                    }

            combined_config = self._combine_best_configs(candidate_configs)
            combined_results = self._test_configuration(
                config=combined_config,
                test_cases=test_cases,
                group_name="combined"
            )
            
            combined_improvement = self._calculate_improvement(
                baseline_results, combined_results
            )
            
            if combined_improvement > best_improvement:
                best_config = {
                    "type": "combined",
                    "config": combined_config,
                    "improvement": combined_improvement,
                    "test_results": combined_results
                }
        
        return best_config
